ES5 Hosting


function someFunction() {

  var j;  // hoisting

  for( var i = 0; i < 4; i++ ) {
    j = i * i;
  }

  console.log( j );
  // > 9

}

Default + Rest + Spread

Default parameters


function add(a, b = 10) {
  return a + b;
}

console.log( add( 3, 5 ) );
// > 8

console.log( add( 3 ) );
// > 13

Rest parameters


function format(message, ...params) {
  for(let p of params) {
    message = message.replace(/\?/, p);
  }
  return message;
}

format( "Die Summe von ? und ? ist ?", 3, 7, 10);
// > Die Summe von 3 und 7 ist 10

Spread parameters


console.info( Math.max(1, 5, 2, 3) );
// > 5

const numbers = [ 1, 5, 2, 3 ];

console.info( Math.max(...numbers) );
// > 5

Destructuring

console.log( text.split(/,/) );
// > ["a", "b", "c", "d", "e"]

const [ first, second ] = text.split(/,/);

console.log( first );
// > "a"

console.log( second );
// > "b"

ES6 Inheritance


class Developer extends Person {

  constructor(name, languages) {
    super(name);
    this.languages = languages;
  }

  getLanguages() {
    return this.languages.join(", ");
  }
}

const christian = new Developer( "Christian",
    [ "Java", "JavaScript" ] );

function* sequence() {

  while( i <= 3 ) {
    yield i;
    i++;
  }
}

let seq = sequence();

seq.next(); // > { value: 1, done: false }

seq.next(); // > { value: undefined, done: true }
