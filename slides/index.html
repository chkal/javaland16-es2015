ES5 Hosting


function someFunction() {

  var j;  // hoisting

  for( var i = 0; i < 4; i++ ) {
    j = i * i;
  }

  console.log( j );
  // > 9

}

ES2015 Block Scope


function someFunction() {

  for(var i = 0; i < 4; i++) {
    let j = i * i;
  }

  console.log(j);
  // > ReferenceError: j is not defined

}
